INCLUDE Irvine32.inc
.data
gap byte " ",0
star byte "*",0
j dword 0
.code
main PROC
mov ecx,5
l1 :
 mov ebp,ecx
 mov ecx,4
 sub ecx,j
 mov edx,offset gap
 cmp ecx,0
 je l4
 l2 :
 call writestring
 loop l2
 l4 :
 mov edx,offset star
 inc j
 mov ecx,j
 l3 :
 call writestring
 loop l3
 mov ecx,ebp
 call crlf
loop l1
exit
main endp
end main
