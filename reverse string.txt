INCLUDE Irvine32.inc
.data
aName BYTE "Muzamil Ali",0
aNamesize=($-aName)-1       ; basic thing in reversing an string
.code 
main PROC
mov ecx,aNamesize
mov esi,0
l1:
	movzx eax,aName[esi]
	push eax   ; for reversing the string we use push coz push decrements
	inc esi
	loop l1
;call crlf
exit
main ENDP
END main
